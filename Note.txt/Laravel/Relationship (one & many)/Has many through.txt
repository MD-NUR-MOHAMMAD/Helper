# ধরা যাক category এর অনেক গুলো pruducts আছে , আবার product এর অনেক image আছে । এখন কোন category তে কত গুলো image আছে তা বের করতে হবে।  
Models --------------------------------------------------------------------------------------------------------------------------------------
class Category extends Model{
  public function Images(){
    return $this->hasManyThrough(Image::class, Product::class);
  }
}
------------------------------------------------------------------------------------------------------------------------------------------------
controller -----------------------------------------------------------------------------------------------------------------------------------
$cat = Category::with('images')->find(1);
